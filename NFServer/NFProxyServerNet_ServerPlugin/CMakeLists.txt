	*.cpp)

	*.h)

	${NFProxyServerNet_ServerPlugin_ROOT_Cpp}
	${NFProxyServerNet_ServerPlugin_ROOT_Hpp})

	LIBRARY_OUTPUT_DIRECTORY ${PROJECT_SOURCE_DIR}/_Out/Comm/ )
